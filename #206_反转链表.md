# 反转链表

> 原题：[https://leetcode-cn.com/problems/reverse-linked-list](https://leetcode-cn.com/problems/reverse-linked-list)

## 描述

反转一个单链表。

## 难度

**简单**

## 示例

```
输入: 1->2->3->4->5
输出: 5->4->3->2->1
```

## 思路

1. 定义一个新的头节点 reverseNode
2. 遍历链表，每遍历一个节点，将其取出，放在新的头节点 reverseNode 的后面
3. 最后返回 reverseNode 的 next 节点

## 代码

### Go

```go
type ListNode struct {
    Val int
    Next *ListNode
}

func reverseList(head *ListNode) *ListNode {
    reverseNode := &ListNode{}
    current := head
    for current != nil {
        // 记录下一个节点
        next := current.Next
        // 将当前节点指向翻转后的第一个节点
        current.Next = reverseNode.Next
        // 修改新的头节点的下一个节点
        reverseNode.Next = current

        // 指向下一个节点
        current = next
    }
    return reverseNode.Next
}

// 打印链表
func printListNode(node *ListNode) {
    var str = ""
    for node != nil {
        str += fmt.Sprintf("%d->", node.Val)
        node = node.Next
    }
    str = strings.TrimSuffix(str, "->")
    fmt.Println(str)
}
```

func TestReverseList(t *testing.T) {

### Java

```java
public class ReverseLinkedList {

    /**
     * 单链表
     */
    public static class ListNode {
        public int val;
        public ListNode next;
        public ListNode(int val) {
            this.val = val;
        }
    }

    public static ListNode reverseList(ListNode head) {
        ListNode reverseNode = new ListNode(0);
        ListNode current = head;
        while (current != null) {
            // 记录下一个节点
            ListNode next = current.next;
            // 将当前节点指向翻转后的第一个节点
            current.next = reverseNode.next;
            // 修改新的头节点的下一个节点
            reverseNode.next = current;

            // 指向下一个节点
            current = next;
        }
        return reverseNode.next;
    }

    private static void printListNode(ListNode node) {
        StringBuilder sb = new StringBuilder();
        while (node != null) {
            sb.append(node.val).append("->");
            node = node.next;
        }
        sb.deleteCharAt(sb.length() - 1);
        sb.deleteCharAt(sb.length() - 1);
        System.out.println(sb.toString());
    }

    public static void main(String[] args) {
        ListNode head = new ListNode(1);
        ListNode node2 = new ListNode(2);
        ListNode node3 = new ListNode(3);
        ListNode node4 = new ListNode(4);
        ListNode node5 = new ListNode(5);
        head.next = node2;
        node2.next = node3;
        node3.next = node4;
        node4.next = node5;

        printListNode(head);

        head = reverseList(head);

        printListNode(head);
    }
}
```

运行结果：

```
1->2->3->4->5
5->4->3->2->1
```
